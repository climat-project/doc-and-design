```clidsl title="sgit.cli"
sgit {
    sub acp(amend a?: flag) {
        action "git add . && git commit $(amend:--amend) && git push $(amend:--force)"
    },
    sub cf(branch: arg, force f: flag) {
        action "git checkout feature/$(branch) $(force:--force)"
    }
}
```
